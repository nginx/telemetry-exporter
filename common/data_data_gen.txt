@namespace("nginx.org") protocol ProductData {
	@df_datatype("ngf") record Data {
		/** The field that identifies what type of data this is. */
		string dataType;
		/** The time the event occurred */
		long eventTime;
		/** The time our edge ingested the event */
		long ingestTime;

	
		
		/** The ProductID of the product. */
		string? ProductID = null;
		
	}
}
